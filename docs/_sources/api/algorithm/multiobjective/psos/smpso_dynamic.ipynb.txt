    "from jmetal.algorithm.multiobjective.smpso import DynamicSMPSO\n",
    "from jmetal.operator import PolynomialMutation\n",
    "from jmetal.problem.multiobjective.fda import FDA2\n",
    "from jmetal.util.archive import CrowdingDistanceArchive\n",
    "from jmetal.util.observable import TimeCounter\n",
    "from jmetal.util.observer import PlotFrontToFileObserver, WriteFrontToFileObserver\n",
    "from jmetal.util.termination_criterion import StoppingByEvaluations\n",
    "\n",
    "problem = FDA2()\n",
    "\n",
    "time_counter = TimeCounter(delay=15)\n",
    "time_counter.observable.register(problem)\n",
    "time_counter.start()\n",
    "\n",
    "max_evaluations = 25000\n",
    "algorithm = DynamicSMPSO(\n",
    "    problem=problem,\n",
    "    swarm_size=100,\n",
    "    mutation=PolynomialMutation(probability=1.0 / problem.number_of_variables, distribution_index=20),\n",
    "    leaders=CrowdingDistanceArchive(100),\n",
    "    termination_criterion=StoppingByEvaluations(max=max_evaluations)\n",
    ")\n",
    "\n",
    "algorithm.observable.register(observer=PlotFrontToFileObserver('front_plot'))\n",
    "algorithm.observable.register(observer=WriteFrontToFileObserver('front_files'))\n",
    "\n",
